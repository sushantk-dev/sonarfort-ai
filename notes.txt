# 1. Build (docker-compose or plain docker)
docker compose build
# or
docker build -t sonarfort-ai:latest .

# 2. Tag for Artifact Registry
docker tag sonarfort-ai:latest \
  REGION-docker.pkg.dev/YOUR_PROJECT_ID/YOUR_REPO/sonarfort-ai:latest

# 3. Authenticate to Artifact Registry
gcloud auth configure-docker REGION-docker.pkg.dev

# 4. Push
docker push REGION-docker.pkg.dev/YOUR_PROJECT_ID/YOUR_REPO/sonarfort-ai:latest
